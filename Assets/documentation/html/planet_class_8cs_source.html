165  {
166  get
167  {
168  // http://www.csun.edu/~hcmth017/master/node16.html
169  float E = m;
170  for (int i = 0; i < 50; i++)
171  {
172  E = (m + e * Mathf.Sin(E));
173  }
174  // https://drive.google.com/file/d/1so93guuhCO94PEU8vFvDLv_-k9vJBcFs/view?usp=sharing
175  float trueAnomaly = 2f * Mathf.Atan(Mathf.Sqrt((1f + e) / (1f - e)) * Mathf.Tan(E / 2f));
176  return trueAnomaly;
177  }
178  }

183  public readonly Vector3 position
184  {
185  get
186  {
187  // http://www.stargazing.net/kepler/ellipse.html#twig04
188  float vpo = v + longOfPeriapsis;
189  float x = r * (Mathf.Cos(longOfAscNode) * Mathf.Cos(vpo) - Mathf.Sin(longOfAscNode) * Mathf.Sin(vpo) * Mathf.Cos(i));
190  float y = r * (Mathf.Sin(longOfAscNode) * Mathf.Cos(vpo) + Mathf.Cos(longOfAscNode) * Mathf.Sin(vpo) * Mathf.Cos(i));
191  float z = r * (Mathf.Sin(vpo) * Mathf.Sin(i));
192 
193  return new Vector3(x, z, y); // switch z and y to match unity coords -> maybe move this elsewhere?

206  public ORBIT(float e, float a, float i, float longOfAscNode, float argOfPeriapsis, float orbitalPeriod)
207  {
208  this.e = e;
209  this.a = a;
210  this.i = i;
211  this.longOfAscNode = longOfAscNode;
212  this.argOfPeriapsis = argOfPeriapsis;
213  this.T = orbitalPeriod;
214 
215  this.periastron = 0;
216 
217  // https://en.wikipedia.org/wiki/Longitude_of_the_periapsis
218  float A = Mathf.Cos(argOfPeriapsis) * Mathf.Cos(longOfAscNode) - Mathf.Sin(argOfPeriapsis) * Mathf.Sin(longOfAscNode) * Mathf.Cos(i);
219  float obliquity = 23.43929111f * Mathf.Deg2Rad;
220  float aa = Mathf.Cos(argOfPeriapsis) * Mathf.Sin(longOfAscNode) + Mathf.Sin(argOfPeriapsis) * Mathf.Cos(longOfAscNode) * Mathf.Cos(i);
221  float B = Mathf.Cos(obliquity) * (aa) - Mathf.Sin(obliquity) * Mathf.Sin(argOfPeriapsis) * Mathf.Sin(i);
222  float C = Mathf.Sin(obliquity) * (aa) + Mathf.Cos(obliquity) * Mathf.Sin(argOfPeriapsis) * Mathf.Sin(i);
223 
224  float alpha = Mathf.Atan(B / A);
225  if (A < 0) alpha += 180 * Mathf.Deg2Rad;
226  float delta = Mathf.Asin(C);
227 
228  this.longOfPeriapsis = Mathf.Atan((Mathf.Sin(alpha) * Mathf.Cos(obliquity) + Mathf.Tan(delta) * Mathf.Sin(delta)) / Mathf.Cos(alpha));
229  if (Mathf.Cos(alpha) < 0) this.longOfPeriapsis += 180 * Mathf.Deg2Rad;
230 
231  this.b = a * Mathf.Sqrt(1 - (e * e));
232  active = true;
233  }
